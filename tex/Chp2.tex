\subsubsection{行间公式}
\begin{equation}
    M=\sum_{k=1}^{K}m_k
\end{equation}

\begin{equation}
    \begin{split}
        f(x,y) & = f(0,0)+\frac{1}{1!}\left(x\pdv{x}+y\pdv{y}\right)f(0,0) \\
        & +\frac{1}{2!}\left(x\pdv{x}+y\pdv{y}\right)^2f(0,0)+K     \\
        & +\frac{1}{n!}\left(x\pdv{x}+y\pdv{y}\right)^xf(0,0)+K
    \end{split}
\end{equation}

\subsubsection{行内公式}
这是一个普通的行内公式$M=\sum_{k=1}^{K}m_k$.
